map<string,set<ll>> mp;
void get_hash(string s,ll idx){
    //cout << s << endl;
    ll n = s.size();
    for (int l = 1; l <= n; l++) {
        for (int i = 0; i <= n - l; i++) {
            string sub = s.substr(i,l);
            //cout << i << " " << l << " " << cur_h << endl;
            mp[sub].insert(idx);
        }
    }
}
