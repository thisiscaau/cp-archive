ll const inf = 1e9 + 7, MAXN = 2e5 + 5;
/* declaration */
ll n,m,pref = 0,count = 0;
ll a[MAXN];
map<ll,ll> mod;
/* workspace */
ll lds(){
    multiset<ll> tail;
    for (int i = 1 ; i <= n ; i++){
        auto it = tail.lower_bound(a[i]);
        if (it == tail.begin()){
            tail.insert(a[i]);
        }
        else {
            it--;
            tail.erase(it);
            tail.insert(a[i]);
        }
    }
    return tail.size();
}

        cin >> a[i]; pref += a[i];
        count += mod[pref % m];
        mod[pref % m]++;
    }
    cout << lds() << endl;
    cout << count;
