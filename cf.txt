#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define fi first
#define pb push_back
#define se second
ll n;
const ll N = 5e5;
ll a[N];
string s;
void update(ll start,ll len){
    vector<ll> res;
    for (int i = 0 ; i <= len ; i++){
        res.pb(a[start+i]);
    }
    reverse(res.begin(),res.end());
    for (int i = 0 ; i <= len ; i++){
        a[start+i] = res[i];
    }
}
signed main(){
    //freopen("PERMUTATION.inp","r",stdin);
    //freopen("PERMUTATION.out","w",stdout);
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin >> n;
    for (int i = 1 ; i <= n ; i++){
        a[i] = i;
    }
    cin >> s;
    for (int i = 0 ; i < s.size() ; i++){
        if (s[i]=='>'&&a[i+1]>a[i+2]){
            continue;
        }
        if (s[i]=='<'&&a[i+1]<a[i+2]){
            continue;
        }
        ll start = i+1, cnt = 0;
        char cur = s[i];
        while (s[i] == cur){
            cnt++;
            i++;
        }
        update(start,cnt);
    }
    for (int i = 1 ; i <= n ; i++){
        cout << a[i] << " ";
    }
}
