#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define fi first
#define se second
ll n;
string s;
ll dp[20][20],p[20];
signed main(){
    //freopen("07.inp","r",stdin);
    //freopen("07.out","w",stdout);
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin >> n >> s;
    p[1] = 1;
    for (int i = 2 ; i <= n ; i++){
        p[i] = p[i-1]*10;
    }
    for (int i = 0 ; i <= n ; i++){
        for (int j = 0 ; j <= n ; j++){
            ll val = s[n*2-i-j] - '0';
            if (i) dp[i][j] = dp[i-1][j] + val*p[i];
            if (j) dp[i][j] = max(dp[i][j],dp[i][j-1]+val*p[j]);
        }
    }
    ll i = n , j = n;
    string res = "";
    while (i || j){
        ll val = s[n*2-i-j] - '0';
        if (i && dp[i][j] == dp[i-1][j] + val*p[i]){
            res += "C";
            i--;
        }
        else {
            res += "K";
            j--;
        }  
    }
    cout << res;
}
