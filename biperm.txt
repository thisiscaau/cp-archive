/* shortcuts */
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define fi first
#define se second
#define pb push_back
#define mp make_pair
typedef pair<ll,ll> ii;
typedef vector<ii> vii;
/* constants */
ll const inf = 1e9 + 7, MAXN = 2e5 + 5;
/* declaration */
ll n,mi,tot,l;
ll mat[2][10005],vis[2][10005],res[2][10005];
/* workspace */
signed main(){
    freopen("BIPERM.inp","r",stdin);
    freopen("BIPERM.out","w",stdout);
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin >> n;
    for (int i = 0 ; i < 2 ; i++){
        for (int j = 1 ; j <= n ; j++){
            cin >> mat[i][j];
            if (!vis[0][mat[i][j]]){
                vis[0][mat[i][j]] = j;
            }
            else {
                vis[1][mat[i][j]] = j;
            }
        }
    }
    tot = 1; mi = 0;
    for (int i = 1 ; i <= n ; i++){
        if (res[0][i] != 0) continue;
        ll idx = i, inv = 0;l = 1;
        res[0][idx] = mat[0][idx]; res[1][idx] = mat[1][idx];
        ll u = idx, v = res[1][idx];
        if (vis[0][v] == idx){
            idx = vis[1][v];
        }
        else {
            idx = vis[0][v];
        }
        while (!res[0][idx]){
            if (mat[0][idx] == res[1][u]){
                res[0][idx] = mat[0][idx];
                res[1][idx] = mat[1][idx];
                l++;
                u = idx;
                v = res[1][idx];
                if (vis[0][v] == idx){
                    idx = vis[1][v];
                }
                else {
                    idx = vis[0][v];
                }
            }
            else {
                res[0][idx] = mat[1][idx];
                res[1][idx] = mat[0][idx];
                l++; inv++;
                u = idx; v = res[1][idx];
                if (vis[0][v] == idx){
                    idx = vis[1][v];
                }
                else {
                    idx = vis[0][v];
                }
            }
        }
        if (l > 1){
            tot *= 2;
        }
        if (inv > l - inv) mi += l - inv;
        else mi += inv;
    }
    cout << tot << " " << mi << endl;
    for (int i = 0 ; i < 2 ; i++){
        for (int j = 1 ; j <= n ; j++){
            cout << res[i][j] << " ";
        }
        cout << endl;
    }
}
